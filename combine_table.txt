----------------------------------------------------------CREATE TABLE----------------------------------------------------------
---------CUSTOMER------------
CREATE TABLE customer(
	customer_id CHAR(4) 		NOT NULL,
 	first_name  VARCHAR(15) 	NOT NULL,
 	last_name   VARCHAR(15) 	NOT NULL,
 	tel_no      CHAR(11) 		CHECK (LENGTH(tel_no) = 11) 	NOT NULL,
 	email       VARCHAR(40) 	CHECK (email LIKE '%@%.%')	NOT NULL,
 	PRIMARY KEY(customer_id)
);

---------FACILITY------------
CREATE TABLE facility(
	facility_id  CHAR(4) 		NOT NULL,
	description  VARCHAR(40) 	NOT NULL,
	max_capacity NUMBER(2) 		NOT NULL,
	location     VARCHAR(5) 	NOT NULL,
	PRIMARY KEY(facility_id)
);

------------SHIP-------------
create table ship (
	ship_id	char(4)	not null,
   	name		varchar(60) not null,
   	max_guest	number(4)	not null,
   	crew_no	number(4)	not null,
   	suites	number(4)	not null,
   	deck 		number(3)	not null,
	primary key(ship_id)
);

----------SUITETYPE----------
CREATE TABLE suitetype (
	type_id 		VARCHAR(5) 		NOT NULL,
 	suite_type 		VARCHAR(50) 	NOT NULL,
 	price 		NUMBER(10, 2)	NOT NULL,
 	max_occupancy 	NUMBER(2)		NOT NULL,
 	suite_size 		VARCHAR(8)		NOT NULL,
 	balcony_size 	VARCHAR(8)		NOT NULL,
  	PRIMARY KEY(type_id)
);

------------ROUTE------------
create table route (
	route_id		char(4)  	not null,
	depature_port	varchar(40) not null,
	destination		varchar(40) not null,
	duration	      varchar(20) not null,
primary key(route_id)
);

-----------OFFERS------------
CREATE TABLE offers(
	offer_code	CHAR(6) 	NOT NULL,
	description VARCHAR(40)	NOT NULL,
	condition	VARCHAR(40)	NOT NULL,
	start_date	DATE		NOT NULL,
	end_date	DATE		NOT NULL,
	discount	NUMBER(3,2)	NOT NULL,
	PRIMARY KEY(offer_code)
);

---------PASSENGER-----------
CREATE TABLE Passenger (
	passenger_id 	CHAR(5) 	NOT NULL,
	first_name 		VARCHAR(50) NOT NULL,
	last_name 		VARCHAR(50) NOT NULL,
	identity_no 	VARCHAR(50) NOT NULL,
	passport_no 	VARCHAR(50) NOT NULL,
	date_of_birth 	DATE 		NOT NULL,
	gender 		CHAR(1) 	NOT NULL CHECK(gender IN ('M', 'F')), 
	telno 		VARCHAR(50) NOT NULL,
	email 		VARCHAR(50) NOT NULL CHECK (email LIKE '%_@__%.__%'),
	address 		VARCHAR(50),
	country 		VARCHAR(20) NOT NULL,
	rating 		NUMBER(2) 	NOT NULL,
	PRIMARY KEY(passenger_id)
);

--------ROUTE_SCHEDULE-------
create table route_schedule (
	schedule_id		char(5)  not null,
   	route_id		char(4) 	not null,
   	ship_id		char(4) 	not null,
   	schedule_date	date 		not null,
   	departure_time   varchar(10) 	not null,
	primary key(schedule_id),
	foreign key(route_id) references route(route_id),
	foreign key(ship_id) references ship(ship_id)
);

-----SUITE-----
create table suite (
  	suite_id		char(4)  not null,
   	type_id		varchar(5) 	not null,
   	ship_id		char(4) 	not null,
   	allocate_deck	number(3) 	not null,
	primary key(suite_id),
	foreign key(type_id) references suitetype(type_id),
	foreign key(ship_id) references ship(ship_id)
);

---------RESERVATION---------
CREATE TABLE reservation(
	reserve_id   CHAR(4) 		NOT NULL,
	date_reserve DATE 		NOT NULL,
 	guest_no     NUMBER(2) 		NOT NULL,
	duration     DECIMAL(3,1) 	NOT NULL,
	facility_id  CHAR(4) 		NOT NULL,
	PRIMARY KEY (reserve_id),
	FOREIGN KEY (facility_id) REFERENCES facility (facility_id)
);

-----------BOOKING-----------
CREATE TABLE booking(
	booking_id		CHAR(5) 	NOT NULL,
	customer_id		CHAR(4) 	NOT NULL,
	schedule_id		CHAR(5) 	NOT NULL,
	booking_date	DATE 		NOT NULL,
	reserve_id		CHAR(4),
	suite_id		CHAR(4)	NOT NULL,
	num_adult		NUMBER(2)	NOT NULL,
	num_child		NUMBER(2)	NOT NULL,
	deposit		NUMBER(6,2)	NOT NULL,
	PRIMARY KEY(booking_id),
	FOREIGN KEY(customer_id) REFERENCES customer(customer_id),
	FOREIGN KEY(schedule_id) REFERENCES route_schedule(schedule_id),
	FOREIGN KEY(reserve_id) REFERENCES reservation(reserve_id),
	FOREIGN KEY(suite_id) REFERENCES suite(suite_id)
);


---------CANCELATION---------
CREATE TABLE cancelation(
	cancelation_id CHAR(4) 		NOT NULL,
	reserve_id     CHAR(4) 		NOT NULL,
	reason         VARCHAR(70) 	NOT NULL,
	penalties      INT 		NOT NULL,
	PRIMARY KEY(cancelation_id),
	FOREIGN KEY (reserve_id) REFERENCES reservation (reserve_id)
);

----------PAYMENT------------
CREATE TABLE payment(
	payment_id		CHAR(4)	NOT NULL,
	booking_id		CHAR(5)	NOT NULL,
	payment_date	DATE		NOT NULL,
	payment_amount	NUMBER(7,2) NOT NULL,
	sst			NUMBER(7,2) NOT NULL,
	service_charge	NUMBER(7,2) NOT NULL,
	offer_code		CHAR(6),
	PRIMARY KEY(payment_id),
	FOREIGN KEY(booking_id) REFERENCES booking(booking_id),
	FOREIGN KEY(offer_code) REFERENCES offers(offer_code)
);


-----PASSENGER_BOOKING-------
create table Passenger_booking (
	passenger_booking_id 	CHAR(5) 	NOT NULL,
	booking_id 			CHAR(5) 	NOT NULL,
	passenger_id 		CHAR(5) 	NOT NULL,
	PRIMARY KEY(passenger_booking_id,passenger_id,booking_id),
	FOREIGN KEY (passenger_id) REFERENCES Passenger(passenger_id),
	FOREIGN KEY (booking_id) REFERENCES booking(booking_id)
);
